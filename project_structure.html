Модели данных

# chepochem_app/models.py

class Role(models.Model):
    name = models.CharField(max_length=50)  # user, moderator, admin

class User(AbstractUser):
    role = models.ForeignKey(Role, on_delete=models.CASCADE)
    phone = models.CharField(max_length=20)
    is_verified = models.BooleanField(default=False)

class Listing(models.Model):
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    category = models.ForeignKey(Category, on_delete=models.CASCADE)
    title = models.CharField(max_length=200)
    description = models.TextField()
    price = models.DecimalField(max_digits=10, decimal_places=2)
    status = models.CharField(max_length=20)  # pending, active, rejected
    created_at = models.DateTimeField(auto_now_add=True)

class Review(models.Model):
    reviewer = models.ForeignKey(User, on_delete=models.CASCADE)
    reviewed_user = models.ForeignKey(User, on_delete=models.CASCADE)
    rating = models.IntegerField()
    comment = models.TextField()
    is_positive = models.BooleanField()

class UserReputation(models.Model):
    user = models.OneToOneField(User, on_delete=models.CASCADE)
    total_score = models.IntegerField(default=0)
    reputation_level = models.CharField(max_length=20)

Хранимые процедуры (5 штук)

-- database/stored_procedures.sql

-- 1. Создание объявления
CREATE OR REPLACE FUNCTION create_listing(
    p_user_id INTEGER,
    p_category_id INTEGER,
    p_title VARCHAR(200),
    p_description TEXT,
    p_price DECIMAL(10,2)
) RETURNS INTEGER AS $$
BEGIN
    INSERT INTO listings (user_id, category_id, title, description, price, status)
    VALUES (p_user_id, p_category_id, p_title, p_description, p_price, 'pending')
    RETURNING id;
END;
$$ LANGUAGE plpgsql;

-- 2. Обновление объявления
CREATE OR REPLACE FUNCTION update_listing(
    p_listing_id INTEGER,
    p_user_id INTEGER,
    p_title VARCHAR(200),
    p_description TEXT,
    p_price DECIMAL(10,2)
) RETURNS BOOLEAN AS $$
BEGIN
    UPDATE listings 
    SET title = p_title, description = p_description, price = p_price
    WHERE id = p_listing_id AND user_id = p_user_id;
    RETURN FOUND;
END;
$$ LANGUAGE plpgsql;

-- 3. Создание отзыва
CREATE OR REPLACE FUNCTION create_review(
    p_reviewer_id INTEGER,
    p_reviewed_user_id INTEGER,
    p_rating INTEGER,
    p_comment TEXT
) RETURNS INTEGER AS $$
BEGIN
    INSERT INTO reviews (reviewer_id, reviewed_user_id, rating, comment, is_positive)
    VALUES (p_reviewer_id, p_reviewed_user_id, p_rating, p_comment, p_rating >= 4)
    RETURNING id;
END;
$$ LANGUAGE plpgsql;

-- 4. Модерация объявления
CREATE OR REPLACE FUNCTION moderate_listing(
    p_listing_id INTEGER,
    p_moderator_id INTEGER,
    p_action VARCHAR(20),
    p_reason TEXT
) RETURNS BOOLEAN AS $$
BEGIN
    UPDATE listings 
    SET status = p_action, moderated_by = p_moderator_id, moderation_reason = p_reason
    WHERE id = p_listing_id;
    RETURN FOUND;
END;
$$ LANGUAGE plpgsql;

-- 5. Удаление объявления
CREATE OR REPLACE FUNCTION delete_listing(
    p_listing_id INTEGER,
    p_user_id INTEGER
) RETURNS BOOLEAN AS $$
BEGIN
    DELETE FROM listings WHERE id = p_listing_id AND user_id = p_user_id;
    RETURN FOUND;
END;
$$ LANGUAGE plpgsql;

Функции БД (7 штук)

-- database/functions.sql

-- 1. Обновление репутации пользователя
CREATE OR REPLACE FUNCTION update_user_reputation(p_user_id INTEGER)
RETURNS VOID AS $$
BEGIN
    UPDATE user_reputation 
    SET total_score = (
        SELECT COALESCE(SUM(rating), 0) FROM reviews 
        WHERE reviewed_user_id = p_user_id
    ),
    reputation_level = CASE 
        WHEN (SELECT COUNT(*) FROM reviews WHERE reviewed_user_id = p_user_id) = 0 THEN 'newbie'
        WHEN (SELECT AVG(rating) FROM reviews WHERE reviewed_user_id = p_user_id) >= 4.5 THEN 'master'
        WHEN (SELECT AVG(rating) FROM reviews WHERE reviewed_user_id = p_user_id) >= 3.5 THEN 'expert'
        ELSE 'trusted'
    END
    WHERE user_id = p_user_id;
END;
$$ LANGUAGE plpgsql;

-- 2. Поиск объявлений
CREATE OR REPLACE FUNCTION search_listings(
    p_search_query VARCHAR(200),
    p_category_id INTEGER,
    p_min_price DECIMAL(10,2),
    p_max_price DECIMAL(10,2)
) RETURNS TABLE(
    id INTEGER,
    title VARCHAR(200),
    price DECIMAL(10,2),
    user_id INTEGER
) AS $$
BEGIN
    RETURN QUERY
    SELECT l.id, l.title, l.price, l.user_id
    FROM listings l
    WHERE l.status = 'active'
    AND (p_search_query IS NULL OR l.title ILIKE '%' || p_search_query || '%')
    AND (p_category_id IS NULL OR l.category_id = p_category_id)
    AND (p_min_price IS NULL OR l.price >= p_min_price)
    AND (p_max_price IS NULL OR l.price <= p_max_price)
    ORDER BY l.created_at DESC;
END;
$$ LANGUAGE plpgsql;

-- 3. Проверка прав доступа
CREATE OR REPLACE FUNCTION check_user_permission(
    p_user_id INTEGER,
    p_permission VARCHAR(100),
    p_entity_type VARCHAR(50),
    p_entity_id INTEGER
) RETURNS BOOLEAN AS $$
DECLARE
    user_role VARCHAR(50);
BEGIN
    SELECT r.name INTO user_role
    FROM users u
    JOIN roles r ON u.role_id = r.id
    WHERE u.id = p_user_id;
    
    -- Проверка прав администратора
    IF user_role = 'admin' THEN
        RETURN TRUE;
    END IF;
    
    -- Проверка прав модератора
    IF user_role = 'moderator' AND p_permission IN ('moderate_listings', 'view_reports') THEN
        RETURN TRUE;
    END IF;
    
    -- Проверка прав владельца ресурса
    IF p_entity_type = 'listing' AND p_entity_id IS NOT NULL THEN
        IF EXISTS(SELECT 1 FROM listings WHERE id = p_entity_id AND user_id = p_user_id) THEN
            RETURN p_permission IN ('edit_own_listing', 'delete_own_listing');
        END IF;
    END IF;
    
    RETURN FALSE;
END;
$$ LANGUAGE plpgsql;

Триггеры (10 штук)

-- database/triggers.sql

-- 1. Автоматическое обновление timestamps
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_listings_updated_at
    BEFORE UPDATE ON listings
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- 2. Создание профиля пользователя
CREATE OR REPLACE FUNCTION create_user_profile()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO user_profiles (user_id, created_at)
    VALUES (NEW.id, CURRENT_TIMESTAMP);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER create_user_profile_trigger
    AFTER INSERT ON users
    FOR EACH ROW
    EXECUTE FUNCTION create_user_profile();

-- 3. Обновление репутации при создании отзыва
CREATE OR REPLACE FUNCTION update_reputation_on_review()
RETURNS TRIGGER AS $$
BEGIN
    PERFORM update_user_reputation(NEW.reviewed_user_id);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_reputation_trigger
    AFTER INSERT ON reviews
    FOR EACH ROW
    EXECUTE FUNCTION update_reputation_on_review();

-- 4. Логирование изменений
CREATE OR REPLACE FUNCTION log_changes()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO audit_log (table_name, operation, old_data, new_data, user_id, created_at)
    VALUES (
        TG_TABLE_NAME,
        TG_OP,
        row_to_json(OLD),
        row_to_json(NEW),
        COALESCE(NEW.user_id, OLD.user_id),
        CURRENT_TIMESTAMP
    );
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER listings_audit_trigger
    AFTER INSERT OR UPDATE OR DELETE ON listings
    FOR EACH ROW
    EXECUTE FUNCTION log_changes();

Система безопасности

RBAC (3 уровня доступа)

  • Пользователь: создание объявлений, отзывы
  • Модератор: модерация, блокировка пользователей
  • Администратор: управление системой, резервные копии

Функции безопасности

  • Шифрование паролей (PBKDF2 + SHA256)
  • Логирование действий пользователей
  • Резервное копирование и восстановление
  • Валидация данных на сервере
